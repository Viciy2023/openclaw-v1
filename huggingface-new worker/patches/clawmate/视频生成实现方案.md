## 需要实现的功能

### 方案A：基于现有图片生成视频（推荐）

**工作流程**：
1. 用户触发视频生成（如"生成一个视频"）
2. 系统先调用现有的图片生成功能
3. 使用生成的图片作为参考，调用视频API（image-to-video）
4. 下载并存储视频文件
5. 返回视频URL给用户

**优势**：
- 保持角色一致性（使用角色参考图）
- 复用现有的姿势选取逻辑
- 视频内容更可控

// 从HTML中提取视频URL
function extractVideoUrl(html: string): string {
  const match = html.match(/src="([^"]+\.mp4)"/);
  return match ? match[1] : '';
}

// 下载视频文件
async function downloadVideo(url: string, savePath: string): Promise<void> {
  const response = await fetch(url);
  const buffer = await response.arrayBuffer();
  await fs.writeFile(savePath, Buffer.from(buffer));
}

// 生成视频保存路径
function buildVideoPath(requestId: string): string {
  const dir = resolveGeneratedVideoDir();
  const fileName = `clawmate-video-${Date.now()}-${requestId}.mp4`;
  return path.join(dir, fileName);
}
